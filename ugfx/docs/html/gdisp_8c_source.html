263 static void hline_clip(GDisplay *g) {
264  // Swap the points if necessary so it always goes from x to x1
265  if (g->p.x1 < g->p.x) {
266  g->p.cx = g->p.x; g->p.x = g->p.x1; g->p.x1 = g->p.cx;
267  }
268 
269  // Clipping
270  #if NEED_CLIPPING
271  #if GDISP_HARDWARE_CLIP == HARDWARE_AUTODETECT
272  if (!g->vmt->setclip)
273  #endif
274  {
275  if (g->p.y < g->clipy0 || g->p.y >= g->clipy1) return;
276  if (g->p.x < g->clipx0) g->p.x = g->clipx0;
277  if (g->p.x1 >= g->clipx1) g->p.x1 = g->clipx1 - 1;
278  if (g->p.x1 < g->p.x) return;
279  }
280  #endif
281

283  // have hardware fills or if we support both hardware pixel drawing and hardware streaming
284  #if GDISP_HARDWARE_FILLS || (GDISP_HARDWARE_DRAWPIXEL && GDISP_HARDWARE_STREAM_WRITE)
285  // Is this a point
286  if (g->p.x == g->p.x1) {
287  drawpixel(g);
288  return;
289  }
290  #endif

293  #if GDISP_HARDWARE_FILLS
294  #if GDISP_HARDWARE_FILLS == HARDWARE_AUTODETECT
295  if (g->vmt->fill)
296  #endif
297  {
298  g->p.cx = g->p.x1 - g->p.x + 1;
299  g->p.cy = 1;

302  }
303  #endif

306  #if GDISP_HARDWARE_FILLS != TRUE && GDISP_HARDWARE_STREAM_POS && GDISP_HARDWARE_STREAM_WRITE
307  #if GDISP_HARDWARE_STREAM_POS == HARDWARE_AUTODETECT
308  if (g->vmt->writepos)
309  #endif

311  if (!(g->flags & GDISP_FLG_SCRSTREAM))
312  setglobalwindow(g);

434 // Parameters: x,y and x1,y1
435 // Alters: x,y x1,y1 cx,cy
436 static void line_clip(GDisplay *g) {
437  int16_t dy, dx;
438  int16_t addx, addy;
439  int16_t P, diff, i;
440 
441  // Is this a horizontal line (or a point)
442  if (g->p.y == g->p.y1) {
443  hline_clip(g);
444  return;
445  }
446 
447  // Is this a vertical line (or a point)
448  if (g->p.x == g->p.x1) {
449  vline_clip(g);
450  return;
451  }
452 
453  // Not horizontal or vertical
454 
455  // Use Bresenham's line drawing algorithm.
456  // This should be replaced with fixed point slope based line drawing
457  // which is more efficient on modern processors as it branches less.
458  // When clipping is needed, all the clipping could also be done up front
459  // instead of on each pixel.
460 
461  if (g->p.x1 >= g->p.x) {
462  dx = g->p.x1 - g->p.x;
463  addx = 1;
464  } else {
465  dx = g->p.x - g->p.x1;
466  addx = -1;
467  }
468  if (g->p.y1 >= g->p.y) {
469  dy = g->p.y1 - g->p.y;
470  addy = 1;
471  } else {
472  dy = g->p.y - g->p.y1;
473  addy = -1;
474  }
475 
476  if (dx >= dy) {
477  dy <<= 1;
478  P = dy - dx;
479  diff = P - dx;
480 
481  for(i=0; i<=dx; ++i) {
482  drawpixel_clip(g);
483  if (P < 0) {
484  P += dy;
485  g->p.x += addx;
486  } else {
487  P += diff;
488  g->p.x += addx;
489  g->p.y += addy;
490  }
491  }
492  } else {
493  dx <<= 1;
494  P = dx - dy;
495  diff = P - dy;
496 
497  for(i=0; i<=dy; ++i) {
498  drawpixel_clip(g);
499  if (P < 0) {
500  P += dx;
501  g->p.y += addy;
502  } else {
503  P += diff;
504  g->p.x += addx;
505  g->p.y += addy;
506  }
507  }
508  }
