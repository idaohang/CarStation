350 #if GDISP_NEED_STREAMING || defined(__DOXYGEN__)
351  /**
352  * @brief Start a streaming operation.
353  * @details Stream data to a window on the display sequentially and very fast.
354  * @note While streaming is in operation - no other calls to GDISP functions
355  * can be made (with the exception of @p gdispBlendColor() and streaming
356  * functions). If a call is made (eg in a multi-threaded application) the other
357  * call is blocked waiting for the streaming operation to finish.
358  * @note @p gdispStreamStop() must be called to finish the streaming operation.
359  * @note If more data is written than the defined area then the results are unspecified.
360  * Some drivers may wrap back to the beginning of the area, others may just
361  * ignore subsequent data.
362  * @note Unlike most operations that clip the defined area to the display to generate
363  * a smaller active area, this call will just silently fail if any of the stream
364  * region lies outside the current clipping area.
365  * @note A streaming operation may be terminated early (without writing to every location
366  * in the stream area) by calling @p gdispStreamStop().
367  *
368  * @param[in] g The display to use
369  * @param[in] x,y The start position
370  * @param[in] cx,cy The size of the streamable area
371  *
372  * @api
373  */
374  void gdispGStreamStart(GDisplay *g, coord_t x, coord_t y, coord_t cx, coord_t cy);
375  #define gdispStreamStart(x,y,cx,cy) gdispGStreamStart(GDISP,x,y,cx,cy)
